function interpolateDaily(data) { let total = data.reduce((acc, daily) => acc + daily.gain, 0); let randomGains = []; let remaining = total; // Generate random gains while maintaining total for (let i = 0; i < data.length - 1; i++) { let gain = Math.random() * remaining; randomGains.push(gain); remaining -= gain; } randomGains.push(remaining); // last element to ensure total correctness // Assign daily gains for (let i = 0; i < data.length; i++) { data[i].gain = randomGains[i]; } }
